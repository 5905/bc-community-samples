SELECT  System.Timestamp as Time, 
     SAMPLE1.Temperature as Temperature1, 
     SAMPLE1.Humidity as HumiditySAMPLE1,
     SAMPLE2.Temperature as Temperature2,
     SAMPLE2.Humidity as HumiditySAMPLE2
 FROM iothub SAMPLE1 TIMESTAMP BY Time 
     JOIN iothub SAMPLE2 TIMESTAMP BY Time 
     ON SAMPLE1.Temperature = SAMPLE2.Temperature 
     AND SAMPLE1.Humidity = SAMPLE2.Humidity
     AND DATEDIFF(ss, SAMPLE1, SAMPLE2) BETWEEN 1 AND 5
 WHERE SAMPLE1.Temperature > 60 AND SAMPLE1.Humidity > 50
